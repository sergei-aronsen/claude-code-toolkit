Key sandbox mechanics the executor must use:

**Fresh install cells use `init-local.sh` (NOT `init-claude.sh`).**
Rationale: `init-claude.sh` fetches from GitHub raw content — unavailable in sandbox. `init-local.sh` installs from the local repo into `.claude/` **relative to CWD**. Sandbox setup:
```bash
CELL_HOME="/tmp/tk-matrix-<cell>-$(date +%s)"
rm -rf "$CELL_HOME"
mkdir -p "$CELL_HOME"
cd "$CELL_HOME"
HOME="$CELL_HOME" bash "$REPO_ROOT/scripts/init-local.sh" --mode <mode>
# Resulting install: $CELL_HOME/.claude/
```

declare -a CELL_WORKTREES=()

sandbox_setup() {
    # sandbox_setup <cell_name> -> echoes CELL_HOME path
    local cell_name="$1"
    local cell_home="/tmp/tk-matrix-${cell_name}-$(date +%s)"
    rm -rf "$cell_home"

stage_sp_cache() {
    # stage_sp_cache <cell_home> [<ver>]
    local cell_home="$1" ver="${2:-5.0.7}"
    local cache_root="$cell_home/.claude/plugins/cache/claude-plugins-official/superpowers/$ver"

# SP code-reviewer agent (stub fixture for matrix cell)
AGENT
}

stage_gsd_cache() {
    # stage_gsd_cache <cell_home>
    local cell_home="$1"
    mkdir -p "$cell_home/.claude/get-shit-done/bin"
    cat > "$cell_home/.claude/get-shit-done/bin/gsd-tools.cjs" <<'GSD'
#!/usr/bin/env node
// Stub GSD fixture for matrix cell
GSD
    chmod +x "$cell_home/.claude/get-shit-done/bin/gsd-tools.cjs"
}

snapshot_foreign_settings() {
    # snapshot_foreign_settings <settings.json> -> echoes JSON of non-TK subtrees
    local settings="$1"
    if [ -f "$settings" ]; then
        jq '{hooks: .hooks, enabledPlugins: .enabledPlugins, user_setting_unrelated: .user_setting_unrelated}' \
            "$settings" 2>/dev/null || echo '{}'
    else
        echo '{}'
    fi
}

seed_foreign_settings() {
    # seed_foreign_settings <cell_home>
    # Write a settings.json with SP/GSD hooks + an unrelated user key to exercise invariant 3.

    mkdir -p "$(dirname "$sj")"
    cat > "$sj" <<'JSON'
{
  "hooks": {
    "PreToolUse": [
      {"matcher": "Bash", "hooks": [{"type": "command", "command": "echo [SP] pretooluse"}]}
    ]
  },
  "enabledPlugins": {
    "superpowers@claude-plugins-official": true
  },
  "user_setting_unrelated": "preserved-by-user",
  "permissions": {"deny": []}

JSON
}

setup_v3x_worktree() {
    # setup_v3x_worktree -> echoes worktree path
    local wt="/tmp/tk-matrix-worktree-$$-$(date +%s)"
    rm -rf "$wt"
    git -C "$REPO_ROOT_ABS" worktree add --detach "$wt" "$PRE_40_COMMIT" >/dev/null 2>&1
    CELL_WORKTREES+=("$wt")
    echo "$wt"
}

cleanup_v3x_worktrees() {
    for wt in "${CELL_WORKTREES[@]:-}"; do
        [ -n "$wt" ] && git -C "$REPO_ROOT_ABS" worktree remove --force "$wt" >/dev/null 2>&1 || true
    done
    CELL_WORKTREES=()
}

trap cleanup_v3x_worktrees EXIT
