From `.planning/phases/04-update-flow/04-RESEARCH.md` §Pattern 5 "V3.x state synthesis":

```bash
synthesize_v3_state() {
    local manifest_file="$1"
    local mode
    mode=$(recommend_mode)
    local installed_csv=""
    while IFS= read -r path; do
        if [[ -f "$CLAUDE_DIR/$path" ]]; then
            if [[ -n "$installed_csv" ]]; then installed_csv+=","; fi
            installed_csv+="$CLAUDE_DIR/$path"   # absolute-ish path so write_state hashes the real file
        fi
    done < <(jq -r '.files | to_entries[] | .value[] | .path' "$manifest_file")
    log_info "First update after v3.x — synthesized install state from filesystem (mode=$mode)."
    write_state "$mode" "$HAS_SP" "$SP_VERSION" "$HAS_GSD" "$GSD_VERSION" "$installed_csv" ""
}
```

From `.planning/phases/04-update-flow/04-RESEARCH.md` §Mode-switch transaction (D-52):

```bash
execute_mode_switch() {
    local new_mode="$1"
    local old_skip new_skip installed
    old_skip=$(compute_skip_set "$STATE_MODE" "$MANIFEST_TMP")
    new_skip=$(compute_skip_set "$new_mode"   "$MANIFEST_TMP")
    installed=$(jq -c '[.installed_files[].path]' <<<"$STATE_JSON")

    # files_to_remove = installed ∩ new_skip   (were installed under old mode, now conflict)
    # files_to_add    = (all_manifest - new_skip) - installed
    local files_to_remove files_to_add all_paths
    all_paths=$(jq -c '[.files | to_entries[] | .value[] | .path]' "$MANIFEST_TMP")
    files_to_remove=$(jq -nc --argjson i "$installed"  --argjson s "$new_skip" \
                          '[$i[] | select(. as $x | $s | index($x) != null)]')
    files_to_add=$(   jq -nc --argjson a "$all_paths"  --argjson s "$new_skip" --argjson i "$installed" \
                          '(($a - $s) - $i)')

    # Remove files (tree backup already snapshotted them in Plan 04-03's step 8)
    while IFS= read -r p; do
        [[ -z "$p" ]] && continue
        rm -f "$CLAUDE_DIR/$p"
        log_info "mode-switch removed: $p"
    done < <(jq -r '.[]' <<<"$files_to_remove")

    # Install files — in this plan, we just stage them; Plan 04-02 owns the download loop.
    # Plan 04-01 records them in an ADD_FROM_SWITCH_JSON global for Plan 04-02 to consume.
    ADD_FROM_SWITCH_JSON="$files_to_add"
    REMOVED_BY_SWITCH_JSON="$files_to_remove"
    STATE_MODE="$new_mode"
    log_info "mode-switch: $STATE_MODE -> $new_mode  (to add: $(jq length <<<"$files_to_add"), removed: $(jq length <<<"$files_to_remove"))"
}
```

**Decision (locked for all Phase 4 plans):** Keep the field name `sha256` that `lib/state.sh::write_state` already writes. Do NOT rename.
