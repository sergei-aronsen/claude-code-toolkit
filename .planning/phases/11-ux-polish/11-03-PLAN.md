#!/usr/bin/env bash
# test-migrate-dry-run.sh — UX-01 SC3 migrate-to-complement.sh --dry-run assertions.
#
# Scenarios (Phase 11 Plan 11-03):
#   1. Duplicates present + --dry-run renders [- REMOVE] header + paths + Total
#   2. Zero filesystem writes (snapshot identical before/after, no backup dir)
#   3. 3-col hash table preserved as diagnostic context
#   4. NO_COLOR=1 strips ANSI
#   5. No duplicates → existing 'No duplicate files found' path unchanged
#
# Usage: bash scripts/tests/test-migrate-dry-run.sh
# Exit:  0 = all pass, 1 = any fail

set -euo pipefail

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
REPO_ROOT="$(cd "${SCRIPT_DIR}/../.." && pwd)"
LIB_DIR="${REPO_ROOT}/scripts/lib"
FIXTURES_DIR="${SCRIPT_DIR}/fixtures"

# Reuse existing manifest fixture when available; fall back to live manifest.
MANIFEST_FIXTURE="${FIXTURES_DIR}/manifest-update-v2.json"
[[ -f "$MANIFEST_FIXTURE" ]] || MANIFEST_FIXTURE="${REPO_ROOT}/manifest.json"

PASS=0
FAIL=0

assert_contains() {
    local needle="$1" haystack="$2" msg="$3"
    if echo "$haystack" | grep -qE -- "$needle"; then
        PASS=$((PASS + 1)); echo "  ✓ ${msg}"
    else
        FAIL=$((FAIL + 1)); echo "  ✗ ${msg}"
        echo "      expected pattern: ${needle}"
    fi
}

assert_not_contains() {
    local needle="$1" haystack="$2" msg="$3"
    if ! echo "$haystack" | grep -qE -- "$needle"; then
        PASS=$((PASS + 1)); echo "  ✓ ${msg}"
    else
        FAIL=$((FAIL + 1)); echo "  ✗ ${msg}"
        echo "      unexpected pattern present: ${needle}"
    fi
}

assert_no_ansi() {
    local haystack="$1" msg="$2"
    if printf '%s' "$haystack" | grep -q $'\x1b\['; then
        FAIL=$((FAIL + 1)); echo "  ✗ ${msg} (ANSI escape found)"
    else
        PASS=$((PASS + 1)); echo "  ✓ ${msg}"
    fi
}

md5_any() {
    if command -v md5 >/dev/null 2>&1; then
        md5 -q "$@"
    else
        md5sum "$@" | awk '{print $1}'
    fi
}

snapshot() {
    local dir="$1"
    if [ ! -d "$dir" ]; then
        echo "empty"
        return
    fi
    find "$dir" -type f -print | sort | while IFS= read -r f; do
        printf '%s %s\n' "$f" "$(md5_any "$f")"
    done | md5_any
}

TMPDIR_ROOT="$(mktemp -d -t tk-migrate-dryrun.XXXXXX)"
trap 'rm -rf "${TMPDIR_ROOT}"' EXIT

# seed_duplicates_in_sandbox <sandbox_home> <count>
# Creates the sandbox HOME, .claude/, .src/, and writes <count> files at
# manifest-skip-set paths so DUPLICATES enumeration finds them.
# Picks the first <count> paths from compute_skip_set output for complement-sp.
seed_duplicates_in_sandbox() {
    local sandbox="$1" want="$2"
    mkdir -p "$sandbox/.claude" "$sandbox/.src"

    # Source compute_skip_set from lib/install.sh in a subshell, query for
    # complement-sp skip-set, then write <want> placeholder files into the sandbox.
    local skip_paths
    skip_paths=$(bash -c '
        source "$1"
        compute_skip_set complement-sp "$2"
    ' -- "$LIB_DIR/install.sh" "$MANIFEST_FIXTURE")

    # Take first $want paths and create them as placeholder files
    echo "$skip_paths" | jq -r '.[]' | head -n "$want" | while IFS= read -r rel; do
        [[ -z "$rel" ]] && continue
        mkdir -p "$sandbox/.claude/$(dirname "$rel")"
        echo "placeholder content" > "$sandbox/.claude/$rel"
    done
}

# run_migrate_dryrun <sandbox> [extra_env_KEY=val...]
# Captures combined output from migrate-to-complement.sh --dry-run.
run_migrate_dryrun() {
    local sandbox="$1"; shift
    local extra_env=("$@")
    env "${extra_env[@]}" \
        TK_MIGRATE_HOME="$sandbox" \
        TK_MIGRATE_LIB_DIR="$LIB_DIR" \
        TK_MIGRATE_MANIFEST_OVERRIDE="$MANIFEST_FIXTURE" \
        TK_MIGRATE_FILE_SRC="$sandbox/.src" \
        TK_MIGRATE_SP_CACHE_DIR="$sandbox/.sp-cache" \
        HAS_SP="true" HAS_GSD="false" \
        SP_VERSION="5.1.0" GSD_VERSION="" \
        bash "$REPO_ROOT/scripts/migrate-to-complement.sh" --dry-run 2>&1 || true
}

# ─────────────────────────────────────────────────
# Scenario 1+2+3: duplicates present → [- REMOVE] group + zero writes + 3-col table
# ─────────────────────────────────────────────────
scenario_remove_group_renders() {
    echo ""; echo "Scenario: [- REMOVE] group renders + 3-col table preserved + zero writes"; echo "---"
    local SCR="${TMPDIR_ROOT}/with-dups"
    seed_duplicates_in_sandbox "$SCR" 3

    local snap_before snap_after OUT
    snap_before=$(snapshot "$SCR/.claude")
    OUT=$(run_migrate_dryrun "$SCR")
    snap_after=$(snapshot "$SCR/.claude")

    assert_contains '^\[- REMOVE\] +[0-9]+ files$' "$OUT" "[- REMOVE] header with right-aligned count"
    assert_contains '^Total: [0-9]+ files$'        "$OUT" "Total: footer present"
    assert_contains 'TK tmpl.+on-disk.+SP equiv'   "$OUT" "3-col table header preserved"

    if [ "$snap_before" = "$snap_after" ]; then
        PASS=$((PASS + 1)); echo "  ✓ zero filesystem writes (snapshot identical)"
    else
        FAIL=$((FAIL + 1)); echo "  ✗ filesystem changed during dry-run"
    fi

    # No backup dir created
    if find "$SCR" -maxdepth 2 -type d -name '.claude-backup-pre-migrate-*' | grep -q .; then
        FAIL=$((FAIL + 1)); echo "  ✗ unexpected backup dir created during dry-run"
    else
        PASS=$((PASS + 1)); echo "  ✓ no backup dir created during dry-run"
    fi
}

# ─────────────────────────────────────────────────
# Scenario: NO_COLOR=1 strips ANSI
# ─────────────────────────────────────────────────
scenario_no_color() {
    echo ""; echo "Scenario: NO_COLOR=1 strips ANSI"; echo "---"
    local SCR="${TMPDIR_ROOT}/nocolor"
    seed_duplicates_in_sandbox "$SCR" 2

    local OUT
    OUT=$(run_migrate_dryrun "$SCR" "NO_COLOR=1")
    assert_no_ansi "$OUT" "NO_COLOR=1: zero ANSI escapes in output"
}

# ─────────────────────────────────────────────────
# Scenario: No duplicates → existing path unchanged
# ─────────────────────────────────────────────────
scenario_no_duplicates() {
    echo ""; echo "Scenario: no duplicates → 'No duplicate files found' path unchanged"; echo "---"
    local SCR="${TMPDIR_ROOT}/no-dups"
    mkdir -p "$SCR/.claude" "$SCR/.src"
    # Empty .claude/ → no files match skip-set → DUPLICATES array is empty

    local OUT
    OUT=$(run_migrate_dryrun "$SCR")
    assert_contains    'No duplicate files found on disk' "$OUT" "no-duplicates message present"
    assert_not_contains '\[- REMOVE\]'                    "$OUT" "[- REMOVE] group NOT printed"
}

# Main
scenario_remove_group_renders
scenario_no_color
scenario_no_duplicates

echo ""; echo "---"
echo "PASS: $PASS"
echo "FAIL: $FAIL"
[[ $FAIL -eq 0 ]] || exit 1
exit 0
