```bash
# strip_sentinel_block <file>
# UN-05 (D-01/D-02/D-03): strip <!-- TOOLKIT-START --> ... <!-- TOOLKIT-END --> blocks
# from <file>, plus exactly ONE leading and ONE trailing blank line around each pair.
# Behavior:
#   - File absent → no-op, return 0
#   - Zero markers (no sentinels) → no-op, return 0 (graceful when CLAUDE.md never had them)
#   - Unmatched markers (start count != end count) → log warning, return 0 (D-02: never partial-strip)
#   - Multiple START/END pairs → strip ALL pairs (D-02 defensive)
#   - Empty result after strip → leave empty file on disk, do NOT delete (D-03 least-destruction)
#
# `local` is permitted here because we are INSIDE a function body.
# RETURN-scoped trap cleans the temp file when the function returns by any path.
strip_sentinel_block() {
    local file="$1"
    [[ -f "$file" ]] || return 0   # absent — nothing to strip

    # Count markers to detect unmatched pairs (D-02 guard).
    # `grep -c` returns 1 with exit 1 on zero matches when -q-style; the `|| true`
    # plus `grep -cF` count form returns 0 with exit 0 cleanly under set -e.
    local starts ends
    starts=$(grep -cF '<!-- TOOLKIT-START -->' "$file" 2>/dev/null || true)
    ends=$(grep -cF '<!-- TOOLKIT-END -->' "$file" 2>/dev/null || true)
    starts="${starts:-0}"
    ends="${ends:-0}"

    if [[ "$starts" -ne "$ends" ]]; then
        log_warning "Unmatched TOOLKIT-START/END markers in $file (starts=$starts, ends=$ends) — leaving file untouched"
        return 0
    fi
    if [[ "$starts" -eq 0 ]]; then
        return 0   # no sentinels — graceful no-op (D-01 strip-only-if-present)
    fi

    # awk strip: for each START/END pair, remove the pair and the surrounding blank
    # line on each side (one before START, one after END). State machine:
    #   in_block         — currently inside a START..END pair → drop line
    #   skip_prev_blank  — just exited a START line; drop the trailing blank if any
    #                      (note: handled at exit-of-block, not entry; see flag swap below)
    #   skip_next_blank  — just exited an END line; drop the next blank if any
    local tmp
    tmp=$(mktemp "${TMPDIR:-/tmp}/sentinel-strip.XXXXXX")
    # shellcheck disable=SC2064  # intentional: capture path at trap-registration time
    trap "rm -f '$tmp'" RETURN

    awk '
        # END marker: leave block, arm trailing-blank skip
        /<!-- TOOLKIT-END -->/   { in_block=0; skip_next_blank=1; next }
        # Inside block: drop everything (including START line catch below)
        in_block                 { next }
        # START marker: enter block, arm leading-blank skip retroactively by
        # buffering the previous-line decision; simpler: just consume the blank
        # we already printed by deferring print one line. Implementation below
        # uses a one-line lookahead via "buf".
        /<!-- TOOLKIT-START -->/ {
            # Drop the previously printed blank if it was blank
            in_block=1
            if (last_was_blank && have_buf) { have_buf=0 }
            else if (have_buf) { print buf; have_buf=0 }
            next
        }
        # Trailing-blank skip after END
        skip_next_blank && /^[[:space:]]*$/ { skip_next_blank=0; next }
        # Otherwise: emit any buffered line, then buffer the current line.
        # last_was_blank tracks if the buffered line is blank (so START can drop it).
        {
            if (have_buf) { print buf }
            buf = $0
            have_buf = 1
            last_was_blank = ($0 ~ /^[[:space:]]*$/) ? 1 : 0
            skip_next_blank = 0
        }
        END {
            if (have_buf && !in_block) { print buf }
        }
    ' "$file" > "$tmp"

    mv "$tmp" "$file"
    log_success "Stripped toolkit sentinel block from $file"
}
```

**IMPORTANT — awk implementation note (lookahead via buffering):** The pattern map's awk skeleton uses two flag-based skips. The version above uses a one-line buffer (`buf`, `have_buf`, `last_was_blank`) so we can retroactively drop the blank line that came BEFORE the START marker. This is the only way to correctly drop the leading blank without two-pass parsing. Test fixture in Plan 03 verifies exact byte-level output.

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| user filesystem (~/.claude/) → uninstall.sh | The script reads CLAUDE.md and runs awk on its content; treats it as data, not code. |
| Symlink at ~/.claude/CLAUDE.md → real file | If an attacker placed a symlink redirecting CLAUDE.md to a system file (e.g., /etc/hosts), the strip could rewrite it. |
| TOCTOU between snapshot and post-check | Between `find ... > SP_SNAP_TMP` (after state read) and `find ... > SP_AFTER_TMP` (at end), a parallel process could mutate SP/GSD. |
| Tempfile race in strip helper | `mktemp + awk + mv` — between mktemp and mv, attacker could swap the tmp file. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-19-02-01 | Tampering | `strip_sentinel_block` rewrites `~/.claude/CLAUDE.md` via `mv tmp file`. If `~/.claude/CLAUDE.md` is a symlink to a system file (e.g., user-attacker placed `ln -s /etc/hosts ~/.claude/CLAUDE.md`), the script would rewrite the target. | accept | Toolkit runs as the user against user-owned `~/.claude/`. The user can already write to `/etc/hosts` only if they have permission; the symlink doesn't grant new privilege. CLAUDE.md location is fixed and well-known; users who symlink it to system files are out of threat scope (self-harm). The mitigation cost (lstat + path validation) exceeds the benefit. Note in operator docs that CLAUDE.md should be a regular file. |
| T-19-02-02 | Tampering | TOCTOU between SP_SNAP_TMP capture and SP_AFTER_TMP capture allows a parallel process to mutate SP/GSD trees, causing false positives (script aborts despite uninstall.sh not being the offender). | accept | Toolkit is a single-user CLI. Concurrent processes mutating `~/.claude/plugins/cache/...` during a `bash uninstall.sh` are exceptional. False positive on this edge case manifests as exit 1 with clear `BUG:` message — the user can re-run after stopping the concurrent process. Acceptable trade-off for defense-in-depth assertion. |
| T-19-02-03 | Tampering | Tempfile race: `mktemp -d` + `awk > tmp` + `mv tmp file`. Between mktemp and mv, attacker could swap or unlink the tmp. | mitigate | `mktemp` returns a path under `${TMPDIR:-/tmp}/sentinel-strip.XXXXXX` with mode 0600 by default. RETURN trap deletes it on function exit. Attacker would need write access to /tmp AND the ability to predict the random suffix in <1ms. Standard Unix tempfile threat model — accepted by all standard tools. |
| T-19-02-04 | Information disclosure | Error message `BUG: ... expected: N files; got: M files` reveals file counts of base-plugin trees. | accept | Counts are non-sensitive (no path leakage, no content). Useful for debugging the rare invariant violation. Not a real disclosure. |

| T-19-02-07 | Elevation of privilege | None — script runs as user, modifies user-owned files, no setuid path. | accept | N/A. |
