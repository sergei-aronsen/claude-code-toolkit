```bash
run_s1() {
    local SANDBOX RC OUTPUT CANARY
    SANDBOX="$(mktemp -d /tmp/test-uninstall-keep-state.XXXXXX)"
    # shellcheck disable=SC2064
    trap "rm -rf '${SANDBOX:?}'" RETURN

    echo "  -- Scenario S1: --keep-state N-choice, state file survives, second run not no-op --"

    if ! (cd "$SANDBOX" && bash "$REPO_ROOT/scripts/init-local.sh" >/dev/null 2>&1); then
        assert_fail "S1 setup: init-local.sh" "install must succeed before test"
        return
    fi

    # Modify a canary to ensure MODIFIED classification on uninstall
    CANARY=$(jq -r '.installed_files[] | .path' "$SANDBOX/.claude/toolkit-install.json" \
             | grep -E '\.(md|json)$' | grep -v 'toolkit-install' | head -1)
    printf '\n# S1 keep-state modification\n' >> "$SANDBOX/$CANARY"

    # First uninstall with --keep-state, answer N to every prompt
    RC=0
    OUTPUT=$(printf 'N\n' | \
        HOME="$SANDBOX" \
        TK_UNINSTALL_HOME="$SANDBOX" \
        TK_UNINSTALL_LIB_DIR="$REPO_ROOT/scripts/lib" \
        TK_UNINSTALL_TTY_FROM_STDIN=1 \
        bash "$REPO_ROOT/scripts/uninstall.sh" --keep-state 2>&1) || RC=$?

    assert_eq "0" "$RC" "S1: first uninstall (--keep-state, N) exits 0"

    # A1: state file still present
    if [ -f "$SANDBOX/.claude/toolkit-install.json" ]; then
        assert_pass "S1-A1: state file preserved after --keep-state run"
    else
        assert_fail "S1-A1: state file preserved" "toolkit-install.json absent after --keep-state"
    fi

    # Second uninstall WITHOUT --keep-state — must not be a no-op
    RC=0
    OUTPUT=$(printf 'y\n' | \
        HOME="$SANDBOX" \
        TK_UNINSTALL_HOME="$SANDBOX" \
        TK_UNINSTALL_LIB_DIR="$REPO_ROOT/scripts/lib" \
        TK_UNINSTALL_TTY_FROM_STDIN=1 \
        bash "$REPO_ROOT/scripts/uninstall.sh" 2>&1) || RC=$?

    # A2: second run proceeded past idempotency guard (backup created)
    assert_contains 'Created backup directory' "$OUTPUT" "S1-A2: second run not a no-op (backup created)"

    # A3: MODIFIED classification was non-empty on second run
    assert_contains 'MODIFIED' "$OUTPUT" "S1-A3: MODIFIED files re-classified on second run"

    # A4: base-plugin diff-q invariant (exit 0 proves it passed)
    assert_eq "0" "$RC" "S1-A4: second run exits 0 (base-plugin diff-q invariant)"
}
```

```markdown
### `--no-banner` (v4.4+)

Pass `--no-banner` (or set `NO_BANNER=1`) to suppress the closing
`To remove: bash <(curl …)` line that both installers print on success. Default behaviour
(flag absent, env unset) is byte-identical to v4.3. Use in CI pipelines or scripted
installs where the banner is redundant noise. Symmetric with `update-claude.sh`, which
already honoured this flag since v4.1.

### `--keep-state` for `uninstall.sh` (v4.4+)

Pass `--keep-state` (or set `TK_UNINSTALL_KEEP_STATE=1`) to preserve
`~/.claude/toolkit-install.json` after the uninstall run. This is a recovery flag
for partial-uninstall sessions: if you answered `N` to every modified-file prompt and
want to re-run the uninstaller to finish the job, the state file must still exist for
re-classification to work. A subsequent `uninstall.sh` run (with or without the flag)
proceeds normally — it is NOT a no-op, because the state file is present.
```

## Shared Patterns

### `NO_BANNER` boolean flag (BANNER-01)

**Source:** `scripts/update-claude.sh` lines 11, 22-24, 1009-1010
**Apply to:** `scripts/init-claude.sh`, `scripts/init-local.sh`

The three-piece pattern used verbatim in both init scripts:

```bash
# (1) default at top of script, before argparse
NO_BANNER=0

# (2) argparse clause (uses 'shift' because init scripts use while/$# loop, not for/arg)
        --no-banner)
            NO_BANNER=1
            shift
            ;;

# (3) gate around the banner echo at end of script
if [[ $NO_BANNER -eq 0 ]]; then
    echo "To remove: bash <(curl -sSL https://raw.githubusercontent.com/sergei-aronsen/claude-code-toolkit/main/scripts/uninstall.sh)"
fi
```

**Critical difference from `update-claude.sh`:** `update-claude.sh` uses `for arg in "$@"` (no `shift`). Both init scripts use `while [[ $# -gt 0 ]]` + `shift`. The argparse clause must include `shift` — copy from `--no-bootstrap)` shape in each file, not from `update-claude.sh:24`.

### Test scaffold (hermetic sandbox)

**Source:** `scripts/tests/test-uninstall.sh` + `scripts/tests/test-uninstall-idempotency.sh`
**Apply to:** `scripts/tests/test-uninstall-keep-state.sh`

```bash
export TK_UNINSTALL_HOME="$SANDBOX"        # redirects CLAUDE_DIR, STATE_FILE, LOCK_DIR
```

stdin injection for modified-file prompts:

```bash
OUTPUT=$(printf 'N\n' | \
    HOME="$SANDBOX" \
```

Use `printf 'N\n'` for "keep" choice, `printf 'y\n'` for "remove" choice. Multiple modified files need one line per file: `printf 'N\nN\nN\n'`.
