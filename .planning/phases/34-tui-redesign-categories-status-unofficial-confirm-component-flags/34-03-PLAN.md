# Plan 34-03: Summary table

## Goal

Print per-entry × per-component status table after dispatch loop completes. Mirrors Phase 25 D-28 contract.

## Tasks

### Task 1: Track per-entry results during dispatch

In dispatch loop, accumulate per-entry × per-component result arrays:

```bash
# Parallel arrays — Bash 3.2 friendly
RESULT_NAMES=()      # entry name
RESULT_MCP_STATE=()  # "installed"|"already"|"failed:<reason>"|"skipped:<reason>"|"na"
RESULT_CLI_STATE=()  # same set
```

Populate after each entry finishes (or is skipped).

### Task 2: Render summary at end

At end of `mcp_wizard_run` (or after dispatch loop in `install.sh`):

```bash
print_integrations_summary() {
  local i name mcp cli notes
  printf '\n${BLUE}━━━ Integrations Install Summary ━━━${NC}\n'
  printf '%-25s %-12s %-12s %s\n' "Entry" "MCP" "CLI" "Notes"
  printf '%-25s %-12s %-12s %s\n' "─────" "───" "───" "─────"
  
  local installed_mcp=0 installed_cli=0 skipped=0 failed=0
  
  for ((i=0; i<${#RESULT_NAMES[@]}; i++)); do
    name="${RESULT_NAMES[$i]}"
    mcp="${RESULT_MCP_STATE[$i]}"
    cli="${RESULT_CLI_STATE[$i]}"
    
    # Glyphs + counters
    case "$mcp" in
      installed) mcp_glyph="${GREEN}✓${NC}"; ((installed_mcp++)) ;;
      already)   mcp_glyph="${CYAN}⊘${NC}" ;;
      failed:*)  mcp_glyph="${RED}✗${NC}"; ((failed++)) ;;
      skipped:*) mcp_glyph="${YELLOW}⊘${NC}"; ((skipped++)) ;;
      na)        mcp_glyph="${NC}—" ;;
    esac
    case "$cli" in
      installed) cli_glyph="${GREEN}✓${NC}"; ((installed_cli++)) ;;
      already)   cli_glyph="${CYAN}⊘${NC}" ;;
      failed:*)  cli_glyph="${RED}✗${NC}"; ((failed++)) ;;
      skipped:*) cli_glyph="${YELLOW}⊘${NC}" ;;
      na)        cli_glyph="${NC}—" ;;
    esac
    
    # Notes: extract reason after colon
    notes=""
    case "$mcp$cli" in
      *failed:*) notes="${mcp#*:}${cli#*:}" | head -c 60 ;;
      *skipped:*) notes="${mcp#*:}${cli#*:}" ;;
    esac
    
    printf '%-25s %-12b %-12b %s\n' "$name" "$mcp_glyph" "$cli_glyph" "$notes"
  done
  
  printf '\n${BOLD}Installed:${NC} %d MCPs, %d CLIs | Skipped: %d | Failed: %d\n' \
    "$installed_mcp" "$installed_cli" "$skipped" "$failed"
}
```
