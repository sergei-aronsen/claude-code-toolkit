### Task 3: Add unofficial confirm prompt

Before invoking `claude mcp add` (or `cli_install`) for a selected entry where `MCP_UNOFFICIAL[$i] == 1`:

```bash
unofficial_confirm() {
  local display="$1"
  local tty_src="${TK_INTEGRATIONS_TTY_SRC:-/dev/tty}"
  
  # Skip if --yes
  if [[ "${ALWAYS_YES:-0}" == "1" ]]; then
    return 0
  fi
  
  # Print warning + prompt
  printf '\n%b! '\''%s'\'' is community-maintained / browser-automation.\n%bInstall anyway? [y/N] ' \
    "${YELLOW}" "$display" "${NC}" >&2
  
  local reply=""
  if [[ -r "$tty_src" ]]; then
    read -r reply < "$tty_src" || reply=""
  else
    reply=""  # fail-closed
  fi
  
  case "$reply" in
    [yY]|[yY][eE][sS]) return 0 ;;
    *) return 1 ;;
  esac
}
```

Call site:

```bash
if [[ "${MCP_UNOFFICIAL[$i]}" == "1" ]]; then
  if ! unofficial_confirm "${MCP_DISPLAY[$i]}"; then
    log_info "Skipping unofficial entry: ${MCP_NAMES[$i]}"
    SKIPPED+=("${MCP_NAMES[$i]} (unofficial declined)")
    continue
  fi
fi
```
