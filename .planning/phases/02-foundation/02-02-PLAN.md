    File structure:

    ```python
    #!/usr/bin/env python3
    """Validate manifest.json v2 schema — path existence, vocabulary, drift, version."""
    import json
    import os
    import sys


    ALLOWED_PLUGINS = {"superpowers", "get-shit-done"}


    def main() -> int:
        if len(sys.argv) != 2:
            print("usage: validate-manifest.py <manifest.json>", file=sys.stderr)
            return 2

        manifest_path = sys.argv[1]
        repo_root = os.path.dirname(os.path.abspath(manifest_path))

        with open(manifest_path) as f:
            m = json.load(f)

        errors = []

        # (d) manifest_version == 2
        mv = m.get("manifest_version")
        if mv != 2:
            errors.append(f"manifest_version must be 2, got {mv!r}")

        listed_command_paths = set()
        listed_skill_paths = set()

        # (a) every files.*[].path exists on disk; (c) conflicts_with vocabulary; entries are dicts
        for section, entries in m.get("files", {}).items():
            for entry in entries:
                if not isinstance(entry, dict):
                    errors.append(f"files.{section} entry is not an object: {entry!r}")
                    continue
                p = entry.get("path", "")
                if not p:
                    errors.append(f"files.{section} entry missing `path`: {entry!r}")
                    continue
                full = os.path.join(repo_root, p)
                if not os.path.exists(full):
                    errors.append(f"path not found on disk: {p}")
                for plugin in entry.get("conflicts_with", []):
                    if plugin not in ALLOWED_PLUGINS:
                        errors.append(
                            f"invalid conflicts_with value {plugin!r} in {p} "
                            f"(allowed: {sorted(ALLOWED_PLUGINS)})"
                        )
                if section == "commands":
                    listed_command_paths.add(p)
                elif section == "skills":
                    listed_skill_paths.add(p)

        # templates.* are objects with path
        for name, entry in m.get("templates", {}).items():
            if not isinstance(entry, dict):
                errors.append(f"templates.{name} is not an object: {entry!r}")
                continue
            p = entry.get("path", "")
            if not p:
                errors.append(f"templates.{name} missing `path`")
                continue
            full = os.path.join(repo_root, p)
            if not os.path.exists(full):
                errors.append(f"templates.{name} path not found on disk: {p}")

        # (b) drift: commands/ files on disk not in manifest
        commands_dir = os.path.join(repo_root, "commands")
    ```
