**Step 3 — Verify per-file invariants** (TEMPLATE-01 + TEMPLATE-02 evidence)

```bash
ERRORS=0
while IFS= read -r f; do
    # Sentinel count = 4
    count=$(grep -cF '<!-- v42-splice:' "$f" 2>/dev/null || true)
    [ "$count" = "4" ] || { echo "FAIL: $f has $count/4 sentinels"; ERRORS=$((ERRORS+1)); }

    # Each named sentinel exactly once
    for name in callout fp-recheck-section output-format-section council-handoff; do
        n=$(grep -cF "<!-- v42-splice: $name -->" "$f" 2>/dev/null || true)
        [ "$n" = "1" ] || { echo "FAIL: $f has $n/1 sentinel '$name'"; ERRORS=$((ERRORS+1)); }
    done

    # Contract markers
    grep -qF 'Council handoff' "$f" || { echo "FAIL: $f missing 'Council handoff'"; ERRORS=$((ERRORS+1)); }
    grep -qF '1. **Read context**' "$f" || { echo "FAIL: $f missing '1. **Read context**'"; ERRORS=$((ERRORS+1)); }

    # Em-dash slot string (U+2014, NOT hyphen-minus, NOT en-dash)
    grep -qF '_pending — run /council audit-review_' "$f" || \
        { echo "FAIL: $f missing em-dash slot string"; ERRORS=$((ERRORS+1)); }
done < <(find templates -path '*/prompts/*.md' \
    \( -name 'SECURITY_AUDIT.md' -o -name 'CODE_REVIEW.md' -o \
       -name 'PERFORMANCE_AUDIT.md' -o -name 'MYSQL_PERFORMANCE_AUDIT.md' -o \
       -name 'POSTGRES_PERFORMANCE_AUDIT.md' -o -name 'DEPLOY_CHECKLIST.md' -o \
       -name 'DESIGN_REVIEW.md' \) | sort)

[ "$ERRORS" -eq 0 ] || { echo "Aborting: $ERRORS verification errors"; exit 1; }
echo "OK: all 49 files satisfy TEMPLATE-01 invariants"
```
