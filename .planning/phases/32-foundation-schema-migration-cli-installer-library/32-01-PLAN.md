**3a — Create `scripts/validate-integrations-catalog.py`** (chmod 755). Hand-written Python 3 stdlib validator per D-10/D-11 / PATTERNS § 3. Use `validate-commands.py` as the template; do NOT import `jsonschema`, `re`, or `pathlib`. Implementation:

```python
#!/usr/bin/env python3
"""validate-integrations-catalog.py — Validate integrations-catalog.json schema (CAT-02 / CAT-03).

Derived from D-10/D-11 (Phase 32 — REQ-IDs CAT-02..03).

Checks performed:
  1. File exists and is valid JSON whose top-level value is an object.
  2. Each entry has display_name (str), category (str), components (object).
  3. category is one of the 10 canonical values (D-08).
  4. At least one of components.mcp / components.cli is present.
  5. components.mcp (if present) has install_args (list of str), env_var_keys
     (list of str), requires_oauth (bool).
  6. components.cli (if present) has detect_cmd (str), install (object) with
     darwin (str) and linux (str). post_install_hint optional.
  7. unofficial (if present) is bool.

Exit 0 + 'integrations-catalog.json validation PASSED (N entries checked)' on stdout.
Exit 1 + per-error lines on stderr + 'integrations-catalog.json validation FAILED (N error(s))' summary.

Usage:
  python3 scripts/validate-integrations-catalog.py                    # validates the canonical file
  python3 scripts/validate-integrations-catalog.py <path>             # validates an arbitrary file
"""

import json
import os
import sys

SCRIPT_DIR = os.path.dirname(os.path.abspath(__file__))
REPO_ROOT = os.path.dirname(SCRIPT_DIR)
DEFAULT_CATALOG = os.path.join(REPO_ROOT, "scripts", "lib", "integrations-catalog.json")

ALLOWED_CATEGORIES = {
    "docs-research", "backend", "payments", "email", "workspace",
    "project-management", "communication", "design", "dev-tools", "monitoring",
}


def fail(message):
    print("ERROR: " + message, file=sys.stderr)


def is_str(v):
    return isinstance(v, str)


def is_bool(v):
    return isinstance(v, bool)


def is_list_of_str(v):
    return isinstance(v, list) and all(isinstance(x, str) for x in v)


def validate_mcp_block(entry_name, mcp):
    errors = 0
    if not isinstance(mcp, dict):
        fail(entry_name + ": components.mcp must be an object")
        return 1
    if "install_args" not in mcp or not is_list_of_str(mcp["install_args"]):
        fail(entry_name + ": components.mcp.install_args missing or not list[str]")
        errors += 1
    if "env_var_keys" not in mcp or not is_list_of_str(mcp["env_var_keys"]):
        fail(entry_name + ": components.mcp.env_var_keys missing or not list[str]")
        errors += 1
    if "requires_oauth" not in mcp or not is_bool(mcp["requires_oauth"]):
        fail(entry_name + ": components.mcp.requires_oauth missing or not bool")
        errors += 1
    return errors


def validate_cli_block(entry_name, cli):
    errors = 0
    if not isinstance(cli, dict):
        fail(entry_name + ": components.cli must be an object")
        return 1
    if "detect_cmd" not in cli or not is_str(cli["detect_cmd"]):
        fail(entry_name + ": components.cli.detect_cmd missing or not str")
        errors += 1
    install = cli.get("install")
    if not isinstance(install, dict):
        fail(entry_name + ": components.cli.install missing or not object")
        errors += 1
    else:
        if "darwin" not in install or not is_str(install["darwin"]):
            fail(entry_name + ": components.cli.install.darwin missing or not str")
            errors += 1
        if "linux" not in install or not is_str(install["linux"]):
            fail(entry_name + ": components.cli.install.linux missing or not str")
            errors += 1
    if "post_install_hint" in cli and not is_str(cli["post_install_hint"]):
        fail(entry_name + ": components.cli.post_install_hint must be str when present")
        errors += 1
    return errors


def validate_entry(name, entry):
    errors = 0
    if not isinstance(entry, dict):
        fail(name + ": entry must be an object")
        return 1
    if "display_name" not in entry or not is_str(entry["display_name"]):
        fail(name + ": display_name missing or not str")
        errors += 1
    if "category" not in entry or not is_str(entry["category"]):
        fail(name + ": category missing or not str")
        errors += 1
    elif entry["category"] not in ALLOWED_CATEGORIES:
        fail(name + ": category '" + entry["category"] + "' not in canonical list (" +
             ", ".join(sorted(ALLOWED_CATEGORIES)) + ")")
        errors += 1
    if "unofficial" in entry and not is_bool(entry["unofficial"]):
        fail(name + ": unofficial must be bool when present")
        errors += 1
    components = entry.get("components")
    if not isinstance(components, dict):
        fail(name + ": components missing or not object")
        return errors + 1
    has_mcp = "mcp" in components
    has_cli = "cli" in components
    if not has_mcp and not has_cli:
        fail(name + ": components must contain at least one of mcp / cli")
        errors += 1
    if has_mcp:
        errors += validate_mcp_block(name, components["mcp"])
    if has_cli:
        errors += validate_cli_block(name, components["cli"])
    return errors


def main():
    catalog_path = sys.argv[1] if len(sys.argv) > 1 else DEFAULT_CATALOG
    if not os.path.isfile(catalog_path):
        fail("catalog not found at: " + catalog_path)
        sys.exit(1)
    try:
        with open(catalog_path, "r", encoding="utf-8") as fh:
            catalog = json.load(fh)
    except json.JSONDecodeError as exc:
        fail("catalog is not valid JSON: " + str(exc))
        sys.exit(1)
    if not isinstance(catalog, dict):
        fail("catalog top-level value must be an object")
        sys.exit(1)

    errors = 0
    for name in sorted(catalog.keys()):
        errors += validate_entry(name, catalog[name])

    if errors > 0:
        print(
            "integrations-catalog.json validation FAILED (" + str(errors) + " error(s))",
            file=sys.stderr,
        )
        sys.exit(1)

    print(
        "integrations-catalog.json validation PASSED (" +
        str(len(catalog)) + " entries checked)"
    )
    sys.exit(0)


if __name__ == "__main__":
    main()
```
